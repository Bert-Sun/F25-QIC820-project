\section{Basic Preliminaries}
\begin{frame}
  \frametitle{A Quick Refresher}
  Given an ensemble \(E \coloneq \{(p_x, \rho_x)\}\) where \(\rho_x \in \Dens(\Cee^m)\),
  find a pair of encoding and decoding quantum channels 
  \[\mce \in \Chan(\Cee^m, \Cee^d), \qquad \mc{D} \in \Chan(\Cee^d, \Cee^m)\]
  for which the composition \(\mc{D} \circ \mce\) acts approximately by the identity on \(E\)
\end{frame}
